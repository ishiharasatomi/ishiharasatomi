## 函数相关
### 防抖与节流
#### 函数防抖
> 防抖：通过setTimeout将事件的多次触发变成一次触发。

函数防抖是为了解决什么问题？
+ 当事件需要被频繁触发时，我们在n秒内延迟它的执行。
+ 当事件在n秒内被触发多次，我们需要重新计时。限制触发次数。
```javascript
// 防抖函数
function debounce(fn, delay, triggerNow) {
    let t = null;
    const debounced = function () {
        const _this = this;
        const args = arguments;
        // 如果存在计时器，清除计时器。避免重复调用
        if (t) {
            clearTimeout(t)
        }

        // 
        if (triggerNow) {
            let exec = !t // 如果时第一次执行, t==null && triggger == true
            // 第一次执行立即调用函数
            t = setTimeout(() => {
                t = null
            })
            if (exec) {
                fn.apply(_this, args)
            } else {
                setTimeout(() => {
                    t = null
                }, delay)
            }
        } else {
            // 需要全部延迟触发的时候
            t = setTimeout(() => {
                fn.apply(_this, args)
            }, delay)
        }

    }
    debounced.remove = function(){
        clearTimeout(t);
        t = null;
    }
    return debounced
}
```
